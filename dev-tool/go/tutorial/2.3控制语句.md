
### for用法

```go
    for i := 1; i <= 10; i++ {
		fmt.Printf(" %d",i)
	}

	i := 0
	for ;i <= 10; {
		fmt.Printf("%d\n", i)
		i += 2
	}

	ii := 0
	for ii <= 10 {
		fmt.Printf("%d\n", ii)
		ii += 2
    }
    
    for no, i := 10, 1; i <= 10 && no <= 19; i, no = i+1, no+1 {

    //break跳出循环
    i:=0

		i++

    //死循环
    for true {
		fmt.Println("hi,world")
```

### switch用法

```go
    a := 1
	switch a {
	case 1:
		fmt.Println("1")
	case 2:
		fmt.Println("2")
    }
    
    switch a := 8;a {
	case 1:
		fmt.Println("1")
	default: 
		fmt.Println("default")
    }
    

    letter := "tcp"
	switch letter {
	case "tcp", "e": //满足这个case没有任何语句需要执行
	case "i":
		fmt.Println("vowel")
	default:
		fmt.Println("not a vowel")
    }
    
    //只会执行满足的第一个case
    num := 75
	switch {
	case num >= 0:
		fmt.Println("aa")
	case num >= 50 && num <= 100:
		fmt.Println("bb")	 
    }
    
    //fallthrough会强制执行后面的 case语句,不管其条件是否满足
    switch {
	case true:
		fmt.Println("2、case 条件语句为 true")
		fallthrough
	case false:
		fmt.Println("3、case 条件语句为 false")
		//fallthrough
	default:
		fmt.Println("6、默认 case")
	}

```

## for range

`for range`可以遍历数组、切片、字符串、map 及通道（channel）

格式:
```go
for key, val := range coll {
    ...
}

for key := range coll {
    ...
}

for key,_ := range coll {
    ...
}

for _, val := range coll {
    ...
}
```

val 始终为集合中对应索引的值拷贝，对它所做的任何修改都不会影响到集合中原有的值

通过 for range 遍历的返回值:

- 数组、切片、字符串返回索引和值。
- map 返回键和值。
- 通道（channel）只返回通道内的值。

根据需要可以使用匿名变量(`_`)来省略key或val,匿名变量本身不会进行空间分配，也不会占用一个变量的名字

```go

func parseStudent() {
	m := make(map[string]*student)
	stus := []student{
		{Name: "zhou", Age: 24},
		{Name: "li", Age: 23},
		{Name: "wang", Age: 22},
	}

	// :=声明一个变量 在for range范围内会复用
	for _, stu := range stus {
		fmt.Println(stu.Name)
		m[stu.Name] = &stu
	}
	fmt.Println("11", m)
}

func parseStudent2() {
	m := make(map[string]*student)
	stus := []student{
		{Name: "zhou", Age: 24},
		{Name: "li", Age: 23},
		{Name: "wang", Age: 22},
	}

	for i, stu := range stus {
		m[stu.Name] = &stus[i]
	}
	fmt.Println("22", m)
}

func TestDemo1A(t *testing.T) {
	parseStudent()
	parseStudent2()
}

func TestForRangeB(t *testing.T) {
	c := make(chan int)
	go func() {
		c <- 1
		c <- 2
		c <- 3
		close(c)  //没有这个for range会报死锁错误
	}()
	for v := range c {
		fmt.Println(v)
	}
}
```
